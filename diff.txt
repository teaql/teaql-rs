+    /// Recursively convert an `EntityGraphNode` into a `GraphNode`,
+    /// preserving comment and operation at each level.
+    fn graph_node_from_entity_graph(
+        &self,
+        eg: EntityGraphNode,
+    ) -> Result<GraphNode, RuntimeError> {
+        let mut node = self.graph_node_from_record(&eg.entity_type, eg.record)?;
+        if let Some(comment) = eg.comment {
+            node.set_comment(comment);
+        }
+        if eg.operation == EntityGraphOperation::Delete {
+            node.operation = GraphOperation::Remove;
+            node.relations.clear();
+            return Ok(node);
+        }
+        for (relation, child) in eg.children {
+            let child_repo = self.scoped_repository(child.entity_type.clone());
+            let child_node = child_repo.graph_node_from_entity_graph(child)?;
+            node.relations.entry(relation).or_default().push(child_node);
+        }
+        Ok(node)
+    }
